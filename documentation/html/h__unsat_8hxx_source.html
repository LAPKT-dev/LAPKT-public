61  virtual void eval( const State& s, float& h_val, std::vector<Action_Idx>& pref_ops ) {
62  h_val = count_goals( s );
63  std::vector< aptk::Action_Idx > app_set;
64  this->problem().applicable_set_v2( s, app_set );
65  for ( unsigned i = 0; i < app_set.size(); i++ ) {
66  int a = app_set[i];
67  const Action& act = *(m_strips_model.actions()[a]);
68  for ( auto g_it = m_strips_model.goal().begin();
69  g_it != m_strips_model.goal().end(); g_it++ ) {
70  if ( !s.entails( *g_it) && act.asserts( *g_it ) ) {
71  pref_ops.push_back( a );
72  break;
73  }
74  }
75  }
76  }

80  float count_goals( const State& s ) const {
81  float c = 0.0f;
82  for ( auto g_it = m_strips_model.goal().begin();
83  g_it != m_strips_model.goal().end(); g_it++ ) {
84  if ( !s.entails( *g_it) ) c += 1.0f;
85  }
86  return c;
87  }
