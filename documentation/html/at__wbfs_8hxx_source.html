38 // Anytime best-first search, with one single open list and one single
39 // heuristic estimator, with delayed evaluation of states generated.
40 // Evaluation function f(n) corresponds to that of Weighted A*
41 // f(n) = g(n) + W h(n)
42 // the value of W decreases each time a solution is found, according to the
43 // value of the decay parameter.
44 
45 template <typename Search_Model, typename Abstract_Heuristic, typename Open_List_Type >
46 class AT_WBFS_SQ_SH : public AT_BFS_SQ_SH<Search_Model, Abstract_Heuristic, Open_List_Type > {
47 
48 public:
49  typedef typename Search_Model::State_Type State;
50  typedef Node< State > Search_Node;
51  typedef Closed_List< Search_Node > Closed_List_Type;
52 
53  AT_WBFS_SQ_SH( const Search_Model& search_problem, float W = 5.0f, float decay = 0.75f )
54  : AT_BFS_SQ_SH<Search_Model, Abstract_Heuristic, Open_List_Type>(search_problem), m_W( W ), m_decay( decay ) {
55  }
56 
57  virtual ~AT_WBFS_SQ_SH() {
58  }

60  virtual void process( Search_Node *head ) {
61  typedef typename Search_Model::Action_Iterator Iterator;
62  Iterator it( this->problem() );
63  int a = it.start( *(head->state()) );
64  while ( a != no_op ) {
65  State *succ = this->problem().next( *(head->state()), a );
66  Search_Node* n = new Search_Node( succ, this->problem().cost( *(head->state()), a ), a, head );
67  if ( this->is_closed( n ) ) {
68  delete n;
69  a = it.next();
70  continue;
71  }
72  n->hn() = head->hn();
73  n->fn() = m_W * n->hn() + n->gn();
74  if( previously_hashed(n) ) {
75  delete n;
76  }
77  else
78  this->open_node(n);
79  a = it.next();
80  }
81 
82  this->inc_eval();
83  }
84 
85  virtual Search_Node* do_search() {
86  Search_Node *head = this->get_node();
87  while(head) {
88  if ( head->gn() >= this->bound() ) {
89  this->inc_pruned_bound();
90  this->close(head);
91  head = this->get_node();
92  continue;
93  }
94 
95  if(this->problem().goal(*(head->state()))) {
96  this->close(head);
97  this->set_bound( head->gn() );
98  m_W *= m_decay;
99  if ( m_W < 1.0f ) m_W = 1.0f;
100  return head;
101  }
102  if ( (time_used() - this->t0() ) > this->time_budget() )
103  return NULL;
104 
105  this->eval( head );
106 
107  this->process(head);
108  this->close(head);
109  head = this->get_node();
110  }
111  return NULL;
112  }
113 
114  virtual bool previously_hashed( Search_Node *n ) {
115  Search_Node *previous_copy = NULL;
116 
117  if( (previous_copy = this->open_hash().retrieve(n)) ) {
118 
119  if(n->gn() < previous_copy->gn())
120  {
121  previous_copy->m_parent = n->m_parent;
122  previous_copy->m_action = n->m_action;
123  previous_copy->m_g = n->m_g;
124  previous_copy->m_f = m_W * previous_copy->m_h + previous_copy->m_g;
125  this->inc_replaced_open();
126  }
127  return true;
128  }
129 
130  return false;
131  }
132 
133 protected:
