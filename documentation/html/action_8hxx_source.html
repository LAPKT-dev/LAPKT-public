125 inline bool Action::possible_supporter( const Action& a1, const Action& a2, Fluent_Vec& pvec )
126 {
127  pvec.clear();
128  for ( unsigned k = 0; k < a1.add_vec().size(); k++ )
129  if ( a2.requires( a1.add_vec()[k] ) )
130  pvec.push_back( a1.add_vec()[k] );
131  return !pvec.empty();
132 }

152 inline bool Action::are_effect_interfering( const Action& a1, const Action& a2 )
153 {
154  for ( unsigned k = 0; k < a1.add_vec().size(); k++ )
155  if ( a2.retracts( a1.add_vec()[k] ) )
156  return true;
157  for ( unsigned k = 0; k < a1.del_vec().size(); k++ )
158  if ( a2.asserts( a1.del_vec()[k] ) )
159  return true;
160  return false;
161 }

183 inline bool Action::can_be_regressed_from( const State& s ) const {
184  // Relevance testing
185  bool relevant = false;
186  for ( unsigned k = 0; k < add_vec().size() && !relevant; k++ )
187  if ( s.entails( add_vec()[k] ) ) relevant = true;
188 
189  for( unsigned i = 0; i < ceff_vec().size() && !relevant; i++ )
190  {
191  for ( unsigned k = 0; k < ceff_vec()[i]->add_vec().size() && !relevant; k++ )
192  if( s.entails( ceff_vec()[i]->add_vec()[k] ) )
193  relevant = true;
194  }
195 
196 
197  if (!relevant) return false;
198  // Now test if no deletes from a are entailed by s
199  for ( unsigned k = 0; k < del_vec().size(); k++ )
200  if ( s.entails( del_vec()[k] ) ) return false;
201 
202  for( unsigned i = 0; i < ceff_vec().size(); i++ )
203  {
204  for ( unsigned k = 0; k < ceff_vec()[i]->del_vec().size(); k++ )
205  if( s.entails( ceff_vec()[i]->del_vec()[k] ) )
206  return false;
207 
208  }
209 
210  return true;
211 }
