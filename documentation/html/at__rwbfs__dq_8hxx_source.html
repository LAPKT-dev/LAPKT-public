38 // Anytime Restarting Weighted Best-First Search, with two open lists - one for preferred operators, the other for non-preferred
39 // operator, and one single heuristic estimator, with delayed evaluation of states generated.
40 // Evaluation function f(n) corresponds to that of Weighted A*
41 // f(n) = g(n) + W h(n)
42 // the value of W decreases each time a solution is found, according to the
43 // value of the decay parameter.
44 //
45 // Details and rationale on how to handle restarts to improve quality of plans found by anytime heuristic search
46 // can be found on the paper:
47 //
48 // The Joy of Forgetting: Faster Anytime Search via Restarting
49 // Richter, S. and Thayer, J. T. and Ruml, W.
50 // ICAPS 2010
51

77  virtual void eval( Search_Node* candidate ) {
78  if ( candidate->seen() ) return;
79  std::vector<Action_Idx> po;
80  this->heuristic().eval( candidate, candidate->hn(), po );
81  for ( unsigned k = 0; k < po.size(); k++ )
82  candidate->add_po( po[k] );
83  }

97  virtual void process( Search_Node *head ) {
98 
99  typedef typename Search_Model::Action_Iterator Iterator;
100  Iterator it( this->problem() );
101  int a = it.start( *(head->state()) );
102  while ( a != no_op ) {
103  State *succ = this->problem().next( *(head->state()), a );
104  Search_Node* n = new Search_Node( succ, this->problem().cost( *(head->state()), a ), a, head, this->problem().num_actions() );
105  bool is_in_closed = in_closed(n);
106  bool is_in_open = in_open(n);
107  bool is_in_seen = in_seen(n);
108  if ( !is_in_closed && !is_in_open && !is_in_seen ) {
109  n->hn() = head->hn();
110  n->fn() = m_W * n->hn() + n->gn();
111 
112  this->open_node(n, head->is_po(a));
113 
114  a = it.next();
115  continue;
116  }
117  if ( is_in_seen ) {
118  Search_Node* n2 = m_seen.retrieve(n);
119  if ( n->gn() < n2->gn() ) {
120  n2->gn() = n->gn();
121  n2->m_parent = n->m_parent;
122  n2->m_action = n->action();
123  }
124  n2->fn() = m_W * n2->hn() + n2->gn();
125  m_seen.erase( m_seen.retrieve_iterator(n2) );
126  this->open_node( n2, n2->parent()->is_po(n2->action()));
127  delete n;
128  a = it.next();
129  continue;
130  }
131  if ( is_in_closed ) {
132  Search_Node* n2 = this->closed().retrieve(n);
133  if ( n->gn() < n2->gn() ) {
134  n2->gn() = n->gn();
135  n2->m_parent = n->m_parent;
136  n2->m_action = n->action();
137  n2->fn() = m_W * n2->hn() + n2->gn();
138  this->closed().erase( this->closed().retrieve_iterator( n2 ) );
139  n2->set_seen();
140  this->open_node( n2, n2->parent()->is_po(n2->action()));
141  }
142  delete n;
143  a = it.next();
144  continue;
145  }
146  if ( is_in_open ) {
147  Search_Node* n2 = this->open_hash().retrieve(n);
148  if ( n->gn() < n2->gn() ) {
149  n2->gn() = n->gn();
150  n2->m_parent = n->m_parent;
151  n2->m_action = n->action();
152  n2->fn() = m_W * n2->hn() + n2->gn();
153  this->inc_replaced_open();
154  }
155  delete n;
156  a = it.next();
157  continue;
158  }
159  assert(false);

189  virtual Search_Node* do_search() {
190  if ( !this->closed().empty() )
191  restart_search();
192  Search_Node *head = this->get_node();
193  while(head) {
194  if ( head->gn() >= this->bound() ) {
195  this->inc_pruned_bound();
196  this->close(head);
197  head = this->get_node();
198  continue;
199  }
200 
201  if(this->problem().goal(*(head->state()))) {
202  this->close(head);
203  this->set_bound( head->gn() );
204  m_W *= m_decay;
205  if ( m_W < 1.0f ) m_W = 1.0f;
206  return head;
207  }
208  if ( (time_used() - this->t0() ) > this->time_budget() )
209  return NULL;
210 
211  this->eval( head );
212 
213  this->process(head);
214  this->close(head);
215 
216  head = this->get_node();
217  }
218  return NULL;
219  }

241  this->closed().clear();
242  // MRJ: Clear the contents of Open
243  this->open_hash().clear();
244  Search_Node *head = this->get_node();
245  while ( head ) {
246  if ( !head->seen() )
247  delete head;
248  else

250  head = this->get_node();
251  }

254  #endif
255 
256  this->open_node( this->root(), false );
257  }

259  virtual bool is_open( Search_Node *n ) {
260  Search_Node *n2 = NULL;
261 
262  if( ( n2 = this->open_hash().retrieve(n)) ) {
263 
264  if(n->gn() < n2->gn())
265  {
266  n2->m_parent = n->m_parent;
267  n2->m_action = n->m_action;
268  n2->m_g = n->m_g;
269  n2->m_f = m_W * n2->m_h + n2->m_g;
270  this->inc_replaced_open();
271  }
272  return true;
273  }
274 
275  return false;
276  }
277 
278  bool is_seen( Search_Node* n ) {
279  Search_Node* n2 = m_seen.retrieve(n);
280 
281  if ( n2 == NULL ) return false;
282 
283  if ( n->gn() < n2->gn() ) {
284  n2->gn() = n->gn();
285  n2->m_parent = n->m_parent;
286  n2->m_action = n->action();
287  }
288  n2->fn() = m_W * n2->hn() + n2->gn();
289  m_seen.erase( m_seen.retrieve_iterator(n2) );
290  this->open_node( n2, n2->parent()->is_po(n2->action()));
291  return true;
292  }
