405  inline void idx2tuple( std::vector<unsigned>& tuple, Fluent_Vec& fl, unsigned idx, unsigned arity ) const
406  {
407  unsigned next_idx, div;
408  unsigned current_idx = idx;
409  int n_atoms = fl.size();
410 
411  for(int i = arity-1; i >= 0 ; i--) {
412  // MRJ: Let's use the fast version
413  // n_atoms = 10
414  // i = 3 (w=4) -> div = 1000
415  // i = 2 (w=3) -> div = 100
416  // i = 1 (w=2) -> div = 10
417  // i = 0 (w=1) -> div = 1
418  div = aptk::unrolled_pow( n_atoms , i );
419 
420  if ( current_idx < div ) {
421  next_idx = current_idx;
422  current_idx = 0;
423  }
424  else {
425  // current index : 32
426  // i = 3 -> next_idx = 32 % 1000 = 32, 32 / 1000 = 0
427  // i = 2 -> next_idx = 32 % 100 = 32, 32 / 100 = 0
428  // i = 1 -> next_idx = 32 % 10 = 2, 32 / 10 = 3
429  // i = 0 -> next_idx = 32 % 1 = 0, 32 / 1 = 32
430  next_idx = current_idx % div;
431  const int div_res = current_idx / div;
432  // if current_idx is zero and is the last index, then take next_idx
433  // i = 3, current_idx = ( 32 / 1000 != 0 || i != 0 ) ? 32 / 1000 : 32 = ( F || T ) ? 0 : 32 = 0
434  current_idx = ( div_res != 0 || i != 0) ? div_res : next_idx;
435  }
436  tuple[ i ] = fl[ current_idx ];
437 
438  current_idx = next_idx;
439  }
