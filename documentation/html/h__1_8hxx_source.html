193  float h_val_bf = eval_func( bf->prec_vec().begin(), bf->prec_vec().end() );
194  float h_val = 0;
195 
196  if ( !bf->asserts(f) ) { // added by conditional effect
197  float min_cond_h = infty;
198  for ( auto ceff : bf->ceff_vec() ) {
199  if ( ceff->asserts( f ) ) {
200  float h_cond = eval_func( ceff->prec_vec().begin(), ceff->prec_vec().end(), h_val_bf );
201  if ( h_cond < min_cond_h )
202  min_cond_h = h_cond;
203  }
204  }
205 
206  assert( !dequal(min_cond_h,infty) );
207  h_val_bf = min_cond_h;
208  }
209 
210  const std::vector<const Action*>& add_acts = m_strips_model.actions_adding( f );
211 
212  for ( unsigned k = 0; k < add_acts.size(); k++ ){
213  const Action* a = add_acts[k];
214  h_val = eval_func( a->prec_vec().begin(), a->prec_vec().end() );
215  if ( !a->asserts( f ) ) { // added by conditional effect
216  float min_cond_h = infty;
217  for ( auto ceff : a->ceff_vec() ) {
218  if ( ceff->asserts(f) ) {
219  float h_cond = eval_func( ceff->prec_vec().begin(), ceff->prec_vec().end(), h_val );

247  float eval_diff( const Best_Supporter& bs ) const {
248  float min_val = infty;
249  if ( bs.act_idx == no_such_index )
250  return 0;
251  const Action* a = m_strips_model.actions()[bs.act_idx];
252  for ( auto p : a->prec_vec() )
253  min_val = std::min( min_val, m_values[p] );
254  if ( bs.eff_idx == no_such_index ) return min_val;
255  for ( auto p : a->ceff_vec()[bs.eff_idx]->prec_vec() )
256  min_val = std::min( min_val, m_values[p] );
257  return min_val;
258  }

261  void update( unsigned p, float v, unsigned act_idx, unsigned eff_idx ) {
262  if ( v > m_values[p] ) return;
263  if ( v >= 0.0f && dequal( v, m_values[p] ) ) {
264  Best_Supporter candidate( act_idx, eff_idx );
265  float candidate_diff = eval_diff( candidate );
266  if ( candidate_diff < m_difficulties[p] ) {
267  m_best_supporters[p] = candidate;
268  m_difficulties[p] = candidate_diff;
269  }
270  return;
271  }
272  m_values[p] = v;
273  if ( !m_already_updated.isset( p ) ) {
274  m_updated.push_back( p );
275  m_already_updated.set( p );
276  }
277  m_best_supporters[p].act_idx = act_idx;
278  m_best_supporters[p].eff_idx = eff_idx;
279  m_difficulties[p] = eval_diff( m_best_supporters[p] );
280  }
