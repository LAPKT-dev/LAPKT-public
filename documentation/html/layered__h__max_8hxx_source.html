106  unsigned eval_diff( const Best_Supporter& bs ) const {
107  unsigned min_val = infinity();
108  if ( bs.act_idx == no_such_index )
109  return 0;
110  const Action* a = m_strips_model.actions()[bs.act_idx];
111  for ( auto p : a->prec_vec() )
112  min_val = std::min( min_val, m_values[p] );
113  if ( bs.eff_idx == no_such_index ) return min_val;
114  for ( auto p : a->ceff_vec()[bs.eff_idx]->prec_vec() )
115  min_val = std::min( min_val, m_values[p] );
116  return min_val;
117  }

119  void update( unsigned p, unsigned v, unsigned act_idx, unsigned eff_idx ) {
120  #ifdef DEBUG_LAYERED_H_MAX
121  std::cout << "Updating: " << m_strips_model.fluents()[p]->signature() << std::endl;
122  #endif
123  if ( m_reached.isset(p) ) {
124  if ( v > m_values[p] ) return;
125  // difficulty
126  Best_Supporter candidate( act_idx, eff_idx );
127  unsigned candidate_diff = eval_diff( candidate );
128  if ( candidate_diff < m_difficulties[p] ) {
129  m_best_supporters[p] = candidate;
130  m_difficulties[p] = candidate_diff;
131  }
132  return;
133  }
134  m_changed = true;
135  m_reached.set(p);
136  m_values[p] = v;
137  m_best_supporters[p].act_idx = act_idx;
138  m_best_supporters[p].eff_idx = eff_idx;
139  m_difficulties[p] = eval_diff( m_best_supporters[p] );
140  #ifdef DEBUG_LAYERED_H_MAX
141  std::cout << "\t v = " << v << std::endl;
142  std::cout << "\t bs: action = " << ( act_idx == no_such_index ? "(init)" : m_strips_model.actions()[act_idx]->signature() );
143  std::cout << "effect = " << ( eff_idx == no_such_index ? -1 : eff_idx) << std::endl;
144  #endif
145  for ( unsigned eff_idx : m_strips_model.relevant_effects( p ) ) {
146  #ifdef DEBUG_LAYERED_H_MAX
147  std::cout << "\t Notifying effect: ";
148  std::cout << "action = " << ( m_strips_model.effects()[eff_idx].act_idx == no_such_index ?
149  "(init)" : m_strips_model.actions()[m_strips_model.effects()[eff_idx].act_idx]->signature() );
150  std::cout << "effect = " << ( m_strips_model.effects()[eff_idx].eff_idx == no_such_index ? -1 : m_strips_model.effects()[eff_idx].eff_idx ) << std::endl;
151  #endif
152  STRIPS_Problem::Trigger& t = m_strips_model.triggers()[eff_idx];
153  t.update( p );
154  if ( t.satisfied() ) {
155  #ifdef DEBUG_LAYERED_H_MAX
156  std::cout << "\t\tActivated!" << std::endl;
157  #endif
158  m_triggered_effects.push_back( eff_idx );
159  }
160  }

163  void compute(const State& s) {
164  unsigned current_level = 0;
165 
166  m_reached.reset();
167  for ( unsigned k = 0; k < m_strips_model.num_fluents(); k++ ) {
168  m_values[k] = m_difficulties[k] = infinity();
169  m_best_supporters[k] = Best_Supporter( no_such_index, no_such_index );
170  }
171 
172  for ( unsigned k = 0; k < m_strips_model.effects().size(); k++ ) {
173  STRIPS_Problem::Trigger& t = m_strips_model.triggers()[k];
174  t.reset();
175  }
176 
177  m_triggered_effects.clear();
178 
179  for ( unsigned p : s.fluent_vec() )
180  update( p, current_level, no_such_index, no_such_index );
181 
182  do {
183  m_changed = false;
184 
185  if ( current_level == 0 ) {
186  // 1. process all actions with empty preconditions (and/or conditional effects)
187  for ( const Action* a : m_strips_model.empty_prec_actions() ) {
188 
189  for ( unsigned a_p : a->add_vec() )
190  update( a_p, current_level + 1, a->index(), no_such_index );
191 
192  for ( unsigned j = 0; j < a->ceff_vec().size(); j++ ) {
193  const Conditional_Effect& ceff = *(a->ceff_vec()[j]);
194  if ( !ceff.prec_vec().empty() ) continue;
195  for ( unsigned a_p : ceff.add_vec() )
196  update( a_p, current_level + 1, a->index(), j );
197  }
198  }
199 
200  }
201 
202  m_current_effects.clear();
203  std::swap( m_triggered_effects, m_current_effects );
204  for ( auto eff_index : m_current_effects ) {
205  #ifdef DEBUG_LAYERED_H_MAX
206  std::cout << "Processing effect: ";
207  std::cout << "action = " << ( m_strips_model.effects()[eff_index].act_idx == no_such_index ?
208  "(init)" : m_strips_model.actions()[m_strips_model.effects()[eff_index].act_idx]->signature() );
209  std::cout << "effect = " << ( m_strips_model.effects()[eff_index].eff_idx == no_such_index ? -1 : m_strips_model.effects()[eff_index].eff_idx ) << std::endl;
210  #endif
211  const Best_Supporter& bs = m_strips_model.effects()[eff_index];
212  /*
213  const Action* a = m_strips_model.actions()[ bs.act_idx ];
214  for ( unsigned a_p : a->add_vec() )
215  update( a_p, current_level + 1, bs );
216  if ( bs.eff_idx == no_such_index ) continue;
217  for ( unsigned a_p : a->ceff_vec()[bs.eff_idx]->add_vec() )
218  update( a_p, current_level + 1, bs );
219  */
220  for ( unsigned a_p : m_strips_model.triggers()[eff_index].effect() )
221  update( a_p, current_level + 1, bs );
222  }
223  #ifdef DEBUG_LAYERED_H_MAX
224  std::cout << "Triggered effects: " << m_triggered_effects.size() << std::endl;
225  #endif
226  current_level++;
227  } while ( m_changed && !goal_reached() );
228 
229  }
