12 /*
13 --------------------------------------------------------------------
14 mix -- mix 3 32-bit values reversibly.
15 For every delta with one or two bits set, and the deltas of all three
16  high bits or all three low bits, whether the original value of a,b,c
17  is almost all zero or is uniformly distributed,
18 * If mix() is run forward or backward, at least 32 bits in a,b,c
19  have at least 1/4 probability of changing.
20 * If mix() is run forward, every bit of c will change between 1/3 and
21  2/3 of the time. (Well, 22/100 and 78/100 for some 2-bit deltas.)
22 mix() was built out of 36 single-cycle latency instructions in a
23  structure that could supported 2x parallelism, like so:
24  a -= b;
25  a -= c; x = (c>>13);
26  b -= c; a ^= x;
27  b -= a; x = (a<<8);
28  c -= a; b ^= x;
29  c -= b; x = (b>>13);
30  ...
31  Unfortunately, superscalar Pentiums and Sparcs can't take advantage
32  of that parallelism. They've also turned some of those single-cycle
33  latency instructions into multi-cycle latency instructions. Still,
34  this is the fastest good hash I could find. There were about 2^^68
35  to choose from. I only looked at a billion or so.
36 --------------------------------------------------------------------
37 */
38 #define jenkins_mix(a,b,c) \
39 { \
40  a -= b; a -= c; a ^= (c>>13); \
41  b -= c; b -= a; b ^= (a<<8); \
42  c -= a; c -= b; c ^= (b>>13); \
43  a -= b; a -= c; a ^= (c>>12); \
44  b -= c; b -= a; b ^= (a<<16); \
45  c -= a; c -= b; c ^= (b>>5); \
46  a -= b; a -= c; a ^= (c>>3); \
47  b -= c; b -= a; b ^= (a<<10); \
48  c -= a; c -= b; c ^= (b>>15); \
49 }

51 /*
52 --------------------------------------------------------------------
53 hash() -- hash a variable-length key into a 32-bit value
54  k : the key (the unaligned variable-length array of bytes)
55  len : the length of the key, counting by bytes
56  initval : can be any 4-byte value
57 Returns a 32-bit value. Every bit of the key affects every bit of
58 the return value. Every 1-bit and 2-bit delta achieves avalanche.
59 About 6*len+35 instructions.
60 
61 The best hash table sizes are powers of 2. There is no need to do
62 mod a prime (mod is sooo slow!). If you need less than 32 bits,
63 use a bitmask. For example, if you need only 10 bits, do
64  h = (h & hashmask(10));
65 In which case, the hash table should have hashsize(10) elements.
66 
67 If you are hashing n strings (ub1 **)k, do it like this:
68  for (i=0, h=0; i<n; ++i) h = hash( k[i], len[i], h);
69 
70 By Bob Jenkins, 1996. bob_jenkins@burtleburtle.net. You may use this
71 code any way you wish, private, educational, or commercial. It's free.
72 
73 See http://burtleburtle.net/bob/hash/evahash.html
74 Use for hash table lookup, or anything where one collision in 2^^32 is
75 acceptable. Do NOT use for cryptographic purposes.
76 --------------------------------------------------------------------
77 */

79 inline ub4 jenkins_hash( ub1* k, ub4 length, ub4 initval)
80 /*k - the key */
81 /*lenth - the length of the key */
82 /*initval - the previous hash, or an arbitrary value */
83 {
84  ub4 a,b,c,len;
85 
86  /* Set up the internal state */
87  len = length;
88  a = b = 0x9e3779b9; /* the golden ratio; an arbitrary value */
89  c = initval; /* the previous hash value */
90 
91  /*---------------------------------------- handle most of the key */
92  while (len >= 12)
93  {
94  a += (k[0] +((ub4)k[1]<<8) +((ub4)k[2]<<16) +((ub4)k[3]<<24));
95  b += (k[4] +((ub4)k[5]<<8) +((ub4)k[6]<<16) +((ub4)k[7]<<24));
96  c += (k[8] +((ub4)k[9]<<8) +((ub4)k[10]<<16)+((ub4)k[11]<<24));
97  jenkins_mix(a,b,c);
98  k += 12; len -= 12;
99  }
100 
101  /*------------------------------------- handle the last 11 bytes */
102  c += length;
103  switch(len) /* all the case statements fall through */
104  {
105  case 11: c+=((ub4)k[10]<<24);
106  case 10: c+=((ub4)k[9]<<16);
107  case 9 : c+=((ub4)k[8]<<8);
108  /* the first byte of c is reserved for the length */
109  case 8 : b+=((ub4)k[7]<<24);
110  case 7 : b+=((ub4)k[6]<<16);
111  case 6 : b+=((ub4)k[5]<<8);
112  case 5 : b+=k[4];
113  case 4 : a+=((ub4)k[3]<<24);
114  case 3 : a+=((ub4)k[2]<<16);
115  case 2 : a+=((ub4)k[1]<<8);
116  case 1 : a+=k[0];
117  /* case 0: nothing left to add */
118  }
119  jenkins_mix(a,b,c);
120  /*-------------------------------------------- report the result */
121  return c;
122 }
