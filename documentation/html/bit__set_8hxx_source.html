223 inline unsigned Bit_Set::min_elem(int lb) const {
224  // https://graphics.stanford.edu/~seander/bithacks.html#ZerosOnRightModLookup
225  auto packs = m_fset.packs();
226  int last_pack = lb/32;
227  int last_mask = (1 << (lb % 32)) - 1;
228  for(unsigned i = last_pack; i < m_fset.npacks(); i++){
229  if(packs[i] != 0){
230  auto p = packs[i] & ~last_mask;
231  if(p != 0)
232  return 32 * i + Mod37BitPosition[(p & -p)%37];
233  //The code above finds the number of zeros that are trailing on
234  //the right, so binary 0100 would produce 2. It makes use of the
235  //fact that the first 32 bit position values are relatively prime
236  //with 37, so performing a modulus division with 37 gives a unique
237  //number from 0 to 36 for each. These numbers may then be mapped
238  //to the number of zeros using a small lookup table.
239  }
240  last_mask = 0;
241  }
242  return m_fset.max_index();
243 }

245 inline unsigned Bit_Set::min_missing(const Bit_Set& other ) const {
246  assert( other.m_fset.max_index() == m_fset.max_index() );
247  auto packs = m_fset.packs();
248  auto opacks = other.m_fset.packs();
249 
250  for(unsigned i = 0; i < m_fset.npacks(); i++){
251  auto p = opacks[i] & ~packs[i];
252  if(p != 0){
253  return 32 * i + Mod37BitPosition[(p & -p)%37];
254  }
255  }
256  return max_index();

264  return bit_count[p%256] +
