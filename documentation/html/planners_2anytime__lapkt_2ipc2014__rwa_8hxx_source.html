356  virtual void process( Search_Node *head ) {

358  head->update_land_graph( m_lgm );
359 
360  std::vector< aptk::Action_Idx > app_set;
361  this->problem().applicable_set_v2( *(head->state()), app_set );
362  for ( unsigned i = 0; i < app_set.size(); i++ ) {
363  int a = app_set[i];
364 
365  State *succ = this->problem().next( *(head->state()), a );
366  Search_Node* n = new Search_Node( succ, this->problem().cost( *(head->state()), a ), a, head, this->problem().num_actions() );
367  bool is_in_closed = in_closed(n);
368  bool is_in_open = in_open(n);
369  bool is_in_seen = in_seen(n);
370 
371  if ( !is_in_closed && !is_in_open && !is_in_seen ) {
372  handle_fresh( head, n, a );
373  continue;
374  }
375 
376  if ( is_in_seen ) {
377  Search_Node* n2 = this->seen().retrieve(n);
378  if ( n->gn() < n2->gn() ) {
379  n2->gn() = n->gn();

386  delete n;
387  continue;
388  }
389  if ( is_in_closed ) {
390  Search_Node* n2 = this->closed().retrieve(n);
391  if ( n->gn() < n2->gn() ) {
392  n2->gn() = n->gn();
393  n2->gn_unit() = n->gn_unit();

397  this->closed().erase( this->closed().retrieve_iterator( n2 ) );
398  handle_seen( head, n2, a );

402  }
