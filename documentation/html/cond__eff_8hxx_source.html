93 inline bool Conditional_Effect::can_be_applied_on( const State& s, bool regress ) const
94 {
95  if ( regress )
96  {
97  // Relevance testing
98  bool relevant = false;
99  for ( unsigned k = 0; k < add_vec().size() && !relevant; k++ )
100  if ( s.entails( add_vec()[k] ) ) relevant = true;
101  if (!relevant) return false;
102  // Now test if no deletes from a are entailed by s
103  for ( unsigned k = 0; k < del_vec().size(); k++ )
104  if ( s.entails( del_vec()[k] ) ) return false;
105  return true;
106  }
107 
108  return s.entails( prec_vec() );
109 }
